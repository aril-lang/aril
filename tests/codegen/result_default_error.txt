--- TITLE ---
Result<T> (default error) lowers to arilrt.Result[T, error]

--- INPUT ---
import fmt
import strconv

// Single-arg Result<int>: the error type defaults to `error`, so the
// emitted Go signature is the full Result[int, error] and a bare stdlib
// error propagates through `try` with no .mapErr bridge.
func parse(raw: string): Result<int> {
  let n = try strconv.atoi(raw)
  return Ok(n * 2)
}

func main() {
  match parse("21") {
    Ok(v)  => fmt.println("ok", v),
    Err(e) => fmt.println("err", e.error()),
  }
}
