--- TITLE ---
HTTP client entry point: http.get(url) is a mechanical (T, error) row → ResultOf(http.Get(url)) yielding the http.Response handle, whose statusCode field is read through the field axis (resp.StatusCode) and whose body drains via io.readAll(resp.body) → ResultOf(io.ReadAll(resp.Body)).

--- INPUT ---
import fmt
import http
import io

func fetch(url: string): int {
  match http.get(url) {
    Ok(resp) => {
      match io.readAll(resp.body) {
        Ok(bytes) => return resp.statusCode,
        Err(_) => return 0,
      }
    },
    Err(_) => return 0,
  }
}

func main() {
  fmt.println(fetch("http://example.com"))
}

type Result[T any, E any] struct {
	Tag uint8
	V   T
	E   E
}

func ResultOk[T any, E any](value T) Result[T, E] {
	return Result[T, E]{Tag: 0, V: value}
}
func ResultErr[T any, E any](err E) Result[T, E] {
	return Result[T, E]{Tag: 1, E: err}
}
func (r Result[T, E]) String() string {
	if r.Tag == 0 {
		return fmt.Sprintf("Ok(%v)", r.V)
	}
	return fmt.Sprintf("Err(%v)", r.E)
}
