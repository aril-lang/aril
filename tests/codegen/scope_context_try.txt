--- TITLE ---
ScopeRef `scope.context` → derived-context var; `try` in a spawn body bails `return <tmp>.E`

--- INPUT ---
import context
import fmt

func work(ctx: context.Context, n: int): Result<int, error> {
  if n < 0 { return Err(error("neg")) }
  return Ok(n * 2)
}

func run(): Result<unit, error> {
  return scope<unit, error> {
    spawn {
      let v = try work(scope.context, 21)
      fmt.println("v", v)
      return Ok(())
    }
  }
}

func main() {
  match run() {
    Ok(_)  => fmt.println("done"),
    Err(e) => fmt.println("err", e),
  }
}
