--- TITLE ---
Option.map / Result.map lower to the free OptionMap / ResultMap helpers

--- INPUT ---
import fmt
import strconv

// `x.map(f)` transforms the Some/Ok payload T -> U (None/Err untouched). It
// lowers to a free helper, not a method, because a Go method cannot introduce
// the fresh U type parameter — the same reason `mapErr` is a free MapErr.
func f(o: Option<int>, r: Result<int, string>) {
  let a = o.map((x) => x > 0)
  let b = r.map((x) => "n=" + strconv.itoa(x))
  fmt.println(a.unwrapOr(false), b.unwrapOr("bad"))
}

func main() {
  f(Some(1), Ok(2))
}

--- GO ---
package main

import (
	"fmt"
	"strconv"
)

type Option[T any] struct {
	Tag uint8
	V   T
}

func OptionSome[T any](value T) Option[T] {
	return Option[T]{Tag: 1, V: value}
}
func OptionNone[T any]() Option[T] {
	return Option[T]{Tag: 0}
}
func (o Option[T]) IsSome() bool { return o.Tag == 1 }
func (o Option[T]) IsNone() bool { return o.Tag == 0 }
func (o Option[T]) UnwrapOr(fallback T) T {
	if o.Tag == 1 {
		return o.V
	}
	return fallback
}

type Result[T any, E any] struct {
	Tag uint8
	V   T
	E   E
}

func ResultOk[T any, E any](value T) Result[T, E] {
	return Result[T, E]{Tag: 0, V: value}
}
func ResultErr[T any, E any](err E) Result[T, E] {
	return Result[T, E]{Tag: 1, E: err}
}
func (r Result[T, E]) IsOk() bool  { return r.Tag == 0 }
func (r Result[T, E]) IsErr() bool { return r.Tag == 1 }
func (r Result[T, E]) UnwrapOr(fallback T) T {
	if r.Tag == 0 {
		return r.V
	}
	return fallback
}
func ResultMap[T any, E any, U any](r Result[T, E], f func(T) U) Result[U, E] {
	if r.Tag == 0 {
		return Result[U, E]{Tag: 0, V: f(r.V)}
	}
	return Result[U, E]{Tag: 1, E: r.E}
}
func OptionMap[T any, U any](o Option[T], f func(T) U) Option[U] {
	if o.Tag == 1 {
		return Option[U]{Tag: 1, V: f(o.V)}
	}
	return Option[U]{Tag: 0}
}
func f(o Option[int], r Result[int, string]) {
	var a = OptionMap(o, func(x int) bool {
		return x > 0
	})
	var b = ResultMap(r, func(x int) string {
		return "n=" + strconv.Itoa(x)
	})
	fmt.Println(a.UnwrapOr(false), b.UnwrapOr("bad"))
}
func main() {
	f(OptionSome(1), ResultOk(2))
}
