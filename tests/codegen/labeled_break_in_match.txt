--- TITLE ---
Labeled break — a `break` inside a `match` arm targets the loop, not the lowered switch

--- INPUT ---
import fmt

func main() {
  var i = 0
  while true {
    match i {
      3 => break,
      _ => { i = i + 1 },
    }
  }
  fmt.println("matched", i)
  var k = 0
  for j in 0..10 {
    if j == 4 { break }
    k = k + 1
  }
  fmt.println("plain", k)
}
