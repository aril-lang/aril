--- TITLE ---
Value-handle in a type annotation: a regexp.Regexp parameter lowers to *regexp.Regexp and keeps its import

--- INPUT ---
import fmt
import regexp

func countMatches(re: regexp.Regexp, inputs: []string): int {
  var n = 0
  for s in inputs {
    if re.matchString(s) {
      n = n + 1
    }
  }
  return n
}

func main() {
  let re = regexp.mustCompile("[0-9]+")
  fmt.println(countMatches(re, ["a1", "bb", "c3", "dd"]))
}

--- GO ---

package main

import (
	"fmt"
	"regexp"
)

func countMatches(re *regexp.Regexp, inputs []string) int {
	var n = 0
	for _, s := range inputs {
		if re.MatchString(s) {
			n = n + 1
		}
	}
	return n
}
func main() {
	var re = regexp.MustCompile("[0-9]+")
	fmt.Println(countMatches(re, []string{"a1", "bb", "c3", "dd"}))
}
