--- TITLE ---
class end-to-end — fields, instance method, static method, implicit `this`

--- INPUT ---
import fmt

class Counter {
  var n: int

  static make(start: int): Counter {
    return Counter(start)
  }

  bump() {
    this.n = this.n + 1
  }

  get(): int {
    return this.n
  }
}

func main() {
  let c = Counter.make(10)
  c.bump()
  c.bump()
  c.bump()
  fmt.println(c.get())
}
