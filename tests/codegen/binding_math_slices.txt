--- GO ---

package main

import (
	"fmt"
	"math"
	"slices"
)

func SlicesReverse[T any](xs []T) []T {
	out := make([]T, len(xs))
	for i, v := range xs {
		out[len(xs)-1-i] = v
	}
	return out
}
func main() {
	fmt.Println(math.Pi > 3.14 && math.Pi < 3.15)
	var xs = []int{5, 3, 8, 1, 8}

}
