--- TITLE ---
Scope nested inside a spawn body — inner scope's return is NOT spawn-converted (inSpawnBody resets at the scope/closure boundary)

--- INPUT ---
import fmt

func main() {
  let ch = makeChannel<int>(1)
  let r = scope<int, error> {
    spawn {
      let inner = scope<int, error> {
        spawn { return Ok(()) }
        15
      }
      match inner {
        Ok(v) => ch.send(v + 1),
        Err(_) => ch.send(0),
      }
      return Ok(())
    }
    ch.recv()
  }
  match r {
    Ok(n) => fmt.println("got:", n),
    Err(e) => fmt.println("err:", e),
  }
}

import (
	"context"
	"fmt"
	"sync"
)

type Result[T any, E any] struct {
	Tag uint8
	V   T
	E   E
}

func ResultOk[T any, E any](value T) Result[T, E] {
	return Result[T, E]{Tag: 0, V: value}
}
func ResultErr[T any, E any](err E) Result[T, E] {
	return Result[T, E]{Tag: 1, E: err}
}
func (r Result[T, E]) String() string {
	if r.Tag == 0 {
		return fmt.Sprintf("Ok(%v)", r.V)
	}
	return fmt.Sprintf("Err(%v)", r.E)
}

type Group struct {
	wg     sync.WaitGroup
	once   sync.Once
	err    error
	cancel context.CancelFunc
}

func NewGroup(parent context.Context) (*Group, context.Context) {
	ctx, cancel := context.WithCancel(parent)
	return &Group{cancel: cancel}, ctx
}

func (g *Group) Go(f func() error) {
	g.wg.Add(1)
	go func() {
		defer g.wg.Done()
		if err := f(); err != nil {
			g.once.Do(func() {
				g.err = err
				g.cancel()
			})
		}
	}()
}

func (g *Group) Wait() error {
	g.wg.Wait()
	g.cancel()
	return g.err
}
