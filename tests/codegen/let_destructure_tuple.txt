--- TITLE ---
`let (a, b) = e` tuple destructuring — bind value to a temp, then components

--- INPUT ---
import fmt

func pair(): (int, string) {
  return (7, "hi")
}

func main() {
  let (n, s) = pair()
  let (a, _) = pair()
  fmt.println(n, s, a)
}
