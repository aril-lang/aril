--- TITLE ---
slices end-to-end — the pure value-view accessors: len, indexing, slicing (no push, D55)

--- INPUT ---
import fmt

func main() {
  let out: []int = [1, 2, 3]
  fmt.println(out.len())
  for i in 0..out.len() {
    fmt.println(out[i])
  }
  let mid = out[1:3]
  fmt.println(mid.len())
}

--- GO ---
package main

import "fmt"

func main() {
	var out []int = []int{1, 2, 3}
	fmt.Println(len(out))
	for i := 0; i < len(out); i++ {
		fmt.Println(out[i])
	}
	var mid = out[1:3]
	fmt.Println(len(mid))
}
